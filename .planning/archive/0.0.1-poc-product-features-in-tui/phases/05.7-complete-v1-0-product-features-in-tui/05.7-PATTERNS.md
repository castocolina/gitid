```go
func Detect(deps Deps) (tool Tool, toolPath string, status AuthStatus) {
    for _, name := range []string{"gh", "glab"} {
        p, err := deps.LookPath(name)
        if err != nil {
            continue
        }
        // probe auth status
        _, code, _ := deps.RunCmd(p, "auth", "status")
        if code == 0 {
            if name == "gh" {
                return ToolGH, p, AuthAuthenticated
            }
            return ToolGLab, p, AuthAuthenticated
        }
        if name == "gh" {
            return ToolGH, p, AuthNotLoggedIn
        }
        return ToolGLab, p, AuthNotLoggedIn
    }
    return 0, "", AuthToolNotFound
}
```

**UploadKey** — arg-slice, no shell, gosec G204 compliant (analog: `tui/deps.go` line 415 `exec.Command("ssh-add", "-l")`):
```go
// UploadKey uploads pubPath to tool. keyType is "authentication" or "signing" for gh;
// "auth" for glab. NEVER uses shell expansion.
func UploadKey(tool Tool, toolPath, pubPath, title, keyType string, deps Deps) (string, error) {
    var args []string
    switch tool {
    case ToolGH:
        // gh ssh-key add <key-file> --title "gitid: <name>" --type authentication
        args = []string{"ssh-key", "add", pubPath, "--title", title, "--type", keyType}
    case ToolGLab:
        // glab ssh-key add <key-file> -t "gitid: <name>" --usage-type auth
        args = []string{"ssh-key", "add", pubPath, "-t", title, "--usage-type", keyType}
    default:
        return "", errors.New("uploader: unknown tool")
    }
    out, code, err := deps.RunCmd(toolPath, args...)
    if err != nil || code != 0 {
        return out, fmt.Errorf("uploader: upload key (%s): exit %d: %w", toolPath, code, err)
    }
    return strings.TrimRight(out, "\n"), nil
}
```

```go
    switch msg := msg.(type) {
    case tea.KeyMsg:
        switch msg.String() {
        case "enter":
            // dispatch adopt cmd
        case "r":
            m.method = adopter.AdoptReferenceInPlace
        case "m":
```

```go
// buildTUIAdopterDeps wires adopter.Deps from real internal packages.
// Mirrors the pattern of buildTUIDeleteDeps in tui/deps.go.
func buildTUIAdopterDeps() adopter.Deps {
    home, _ := os.UserHomeDir()
    gitconfigPath := filepath.Join(home, ".gitconfig")
    gitconfigDDir := filepath.Join(home, ".gitconfig.d")

    return adopter.Deps{
        ReadFile: func(path string) ([]byte, error) {
            return os.ReadFile(path) //nolint:gosec // gitid-managed path (G304)
        },
        WriteFile: func(path string, content []byte, mode os.FileMode) (string, error) {

func buildTUIUploaderDeps() uploader.Deps {
    return uploader.Deps{
        LookPath: exec.LookPath,
        RunCmd: func(name string, args ...string) (string, int, error) {
            cmd := exec.Command(name, args...) //nolint:gosec // arg-slice; no shell (G204)
            out, err := cmd.CombinedOutput()
            output := string(out)
            if err == nil {
                return output, 0, nil
            }
            var exitErr *exec.ExitError
            if errors.As(err, &exitErr) {
                return output, exitErr.ExitCode(), nil
            }
            return "", 2, err
        },
    }
```

```go
// liveIncludeIfPreview renders the includeIf block string for the current
// strategy and field values. Called on every keystroke to update the preview
// pane (5.6 D-06). Uses gitconfig.RenderIncludeIf — never hand-builds conditions.
func liveIncludeIfPreview(strategy matchStrategy, name, alias, gitdirVal, hasconfigVal string) string {
    fragPath := "~/.gitconfig.d/" + name
    var matches []gitconfig.Match
    switch strategy {
    case strategyGitdir:
        matches = []gitconfig.Match{{Kind: gitconfig.MatchGitdir, Value: gitdirVal}}
    case strategyHasconfig:
        matches = []gitconfig.Match{{Kind: gitconfig.MatchHasconfig, Value: "remote.*.url:" + hasconfigVal}}
    case strategyBoth:
        matches = []gitconfig.Match{
            {Kind: gitconfig.MatchGitdir, Value: gitdirVal},
            {Kind: gitconfig.MatchHasconfig, Value: "remote.*.url:" + hasconfigVal},
        }
    }
    return gitconfig.RenderIncludeIf(name, fragPath, matches)
}

// defaultHasconfigPattern derives the canonical hasconfig pattern from the alias (D-03).
// Recipe form: git@<alias>:*/**
func defaultHasconfigPattern(alias string) string {
    if alias == "" {
        return ""
    }
    return "git@" + alias + ":*/**"
}
```

```go
// FakeGHDir writes a mode-switching fake gh script and sets GITID_FAKE_GH_MODE.
// The caller prepends the returned dir to PATH via cmd.Env.
// Modes: "ok" (auth ok + upload succeeds), "auth-fail" (not logged in),
//        "not-found" (caller must NOT prepend this dir — simulates gh absent).
//
// Script is a static literal — never constructed from user input (G204-clean).
func FakeGHDir(t *testing.T, mode string) string {
    t.Helper()
    dir := t.TempDir()
    const script = "#!/bin/sh\n" +
        "case \"$1\" in\n" +
        "  auth)\n" +
        "    case \"$GITID_FAKE_GH_MODE\" in\n" +
        "      ok) exit 0 ;;\n" +
        "      *) echo \"error: not logged into github.com\"; exit 1 ;;\n" +
        "    esac\n" +
        "    ;;\n" +
        "  ssh-key)\n" +
        "    case \"$GITID_FAKE_GH_MODE\" in\n" +
        "      ok) echo \"Added SSH key.\"; exit 0 ;;\n" +
        "      *) echo \"error: not authenticated\"; exit 1 ;;\n" +
        "    esac\n" +
        "    ;;\n" +
        "esac\n"
    scriptPath := filepath.Join(dir, "gh")
    if err := os.WriteFile(scriptPath, []byte(script), 0o700); err != nil { //nolint:gosec // test-only static script (G306)
        t.Fatalf("FakeGHDir: writing fake gh: %v", err)
    }
    t.Setenv("GITID_FAKE_GH_MODE", mode)
    return dir
}
```
